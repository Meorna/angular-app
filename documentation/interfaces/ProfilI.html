export interface ProfilI {
    _id?:any;
    nom:string;
    prenom:string;
    specialites:Array<string>;

    mail?:string;
    age:number;
    photo?:string;
    statut?:number;
    token?:string;
}

  _id = 0;

  specialites = [];
  promo = "";
  mail = "";
  age = 0;
  photo = "";
  statut = 0;
}
